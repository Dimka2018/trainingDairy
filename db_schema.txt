CREATE TABLE `EXERCISE` (
	`_ID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`NAME`	TEXT NOT NULL UNIQUE
);

CREATE TABLE `MASS_TYPE` (
	`_ID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`NAME`	INTEGER NOT NULL UNIQUE
);

CREATE TABLE `PROFILE` (
	`_id`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`NAME`	TEXT NOT NULL UNIQUE,
	`IS_ACTIVE`	INTEGER NOT NULL DEFAULT 0,
	`CURRENT_DAY_ID`	INTEGER
);

CREATE TABLE `TRAINEE` (
	`_ID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`PROFILE_ID`	INTEGER NOT NULL,
	FOREIGN KEY(`PROFILE_ID`) REFERENCES `PROFILE`(`_id`) ON DELETE CASCADE
);

CREATE TABLE `HISTORY` (
	`_ID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`IS_COMPLETED`	INTEGER NOT NULL DEFAULT 0,
	`IS_MAX_WEIGHT`	INTEGER NOT NULL DEFAULT 0,
	`WEIGHT`	INTEGER NOT NULL DEFAULT 0,
	`NUMBER_REPS`	INTEGER NOT NULL DEFAULT 0,
	`NUMBER_SETS`	INTEGER NOT NULL DEFAULT 0,
	`MASS_TYPE_ID`	INTEGER NOT NULL,
	`EXERCISE_ID`	INTEGER NOT NULL,
	`TRAINEE_ID`	INTEGER NOT NULL,
	FOREIGN KEY(`TRAINEE_ID`) REFERENCES `TRAINEE`(`_ID`) ON DELETE CASCADE
);

CREATE TABLE `TRAINING_PART` (
	`_ID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`DAY_ID`	INTEGER NOT NULL,
	`EXERCISE_ID`	INTEGER NOT NULL,
	`NUMBER_SETS`	INTEGER NOT NULL DEFAULT 0,
	`NUMBER_REPS`	INTEGER NOT NULL DEFAULT 0,
	`LAST_HISTORY_ID`	INTEGER,
	FOREIGN KEY(`DAY_ID`) REFERENCES `DAY`(`_ID`) ON DELETE CASCADE
);

CREATE TABLE `WORKOUT` (
	`_ID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`NAME`	TEXT NOT NULL,
	`PROFILE_ID`	INTEGER NOT NULL,
	`IS_ACTIVE`	INTEGER NOT NULL DEFAULT 0,
	FOREIGN KEY(`PROFILE_ID`) REFERENCES `PROFILE`(`_id`) ON DELETE CASCADE
);

CREATE TABLE `DAY` (
	`_ID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`WORKOUT_ID`	INTEGER NOT NULL,
	FOREIGN KEY(`WORKOUT_ID`) REFERENCES `WORKOUT`(`_ID`) ON DELETE CASCADE
);

CREATE TABLE `sqlite_sequence` (
	`name`	TEXT,
	`seq`	TEXT
);
